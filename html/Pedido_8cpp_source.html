87bool Pedido::pedidosConcorrentes(const Pedido &pedido) {
88 switch (pedido.tipo) {

91 if (this->estudante.getStudentNumber() == pedido.estudante.getStudentNumber() && this->uc == pedido.uc) {
92 std::cout << "Já existe um pedido para remover o estudante " << estudante.getStudentNumber() << " da UC " << uc << std::endl;
93 return false;
94 }
95 break;
96 }
97 case TipoAlteracao::H: {
98 // Se um dos estudantes tiver um pedido de remoção desta UC em específico não podem trocar
99 if (this->tipo == TipoAlteracao::R) {
100 if (this->estudante.getStudentNumber() == pedido.estudante.getStudentNumber() && this->uc == pedido.uc) {
101 std::cout << "O estudante " << pedido.estudante.getStudentNumber() << " tem um pedido para remover a UC " << this->uc << std::endl;
102 return false;
103 }
104 if (this->estudante.getStudentNumber() == pedido.outroEstudante.getStudentNumber() && this->uc == pedido.uc) {
105 std::cout << "O estudante " << pedido.outroEstudante.getStudentNumber() << " tem um pedido para remover a UC " << this->uc << std::endl;
106 return false;

108 }
109 break;
110 }
