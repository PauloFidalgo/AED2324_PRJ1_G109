742bool Manager::trocaValida(Pedido &pedido) const {
743 Estudante estudante = pedido.getEstudante();
744 Estudante outro = pedido.getOutroEstudante();
745 std::string uc = pedido.getUc();
746
747 // Verifica se os dois estudantes esto inscritos na UC
748 if (!(estudante.inscrito(uc) && outro.inscrito(uc))) {
749 // Pelo menos um dos alunos no esta inscrito na UC
750 cout << string(52,'-') << endl;
751 cout << "| Troca invlida, a Unidade curricular no comum |" << endl;
752 cout << string(52,'-') << endl;
753 return false;
754 }
755
756 std::string turma = estudante.getTurma(uc);
757 std::string outraTurma = outro.getTurma(uc);
758
759 // Verifica se os dois estudantes j so da mesma turma
760 if (turma == outraTurma) {
761 // J so de turmas iguais
762 cout << string(47,'-') << endl;
763 cout << "| Troca invlida, j so da mesma turma |" << endl;
764 cout << string(47,'-') << endl;
765 return false;
766 }
767
